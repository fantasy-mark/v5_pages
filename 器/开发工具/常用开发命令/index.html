    常用开发命令

    bash小技巧

    mkdir very_long_dirname
# !^/!$ 指代 第一/最后 参数, !:n 指代第n个参数(从1开始)
# 比如打错了 mvv one_dir other_dir
mv !^ !$

# 执行上条命令, 上上条则 !-2
!!
# 执行 history 对应 index 命令
!1152

# ctrl + r 搜索历史命令
# ctrl + a 光标移动最前
# ctrl + e 光标移动最后
# ctrl + h 删除光标前面一个字符

    split/cat - 文件分割、组合

    split -b 50m all.tar part_
cat part_* > all.tar

    find - 搜索

    find . -name "*.py" | xargs grep "print(path)"

    curl - 下载

    # 断点续传下载文件, 需要服务器解析Content-Range请求头
curl -C - -O https://releases.ubuntu.com/22.04/ubuntu-22.04-desktop-amd64.iso
# Restful 请求
curl -X POST --data 'keyword=value' http://domain.com/mypath/
# 添加头部
curl -H 'Content-Type:application/json' -H 'Authorization: bearer MyToken' http://domain.com/mypath/
# 设置代理
curl -x socks5://proxyuser:proxypassword@proxy.domain.com:8001 https://domain.com/mypath/
# 帮助
curl --help all

    cron - 定时任务

    servicec cron start
service cron status
 * cron is running

cat /etc/cron.d/daily-cron
# 每3分钟执行一次
# */3 * * * * date >> /var/log/cron.log 2>&1
# 每3小时执行一次
# 0 */3 * * * cd /code && python3 daily.py
# 编辑daily-cron后必须执行以下命令
crontab /etc/cron.d/daily-cron 
service cron reload
service cron restart

    git命令

    • 配置
    # add文件显示乱码
git config --global core.quotepath false core.quotepath=false
# LF will be replaced by CRLF the next time Git touches it
git config --global core.autocrlf true

    配置提交信息

    # 本项目
git config --local user.email "you@example.com"
git config --local user.name "Your Name"

# 全局
git config --global user.email "hacker.do@163.com"
git config --global user.name "Mark"
    • 新建新仓库
    git clone https://jihulab.com/1987/mydocs.git
cd mydocs
git switch -c main
touch README.md
git add README.md
git commit -m "add README"
git push -u origin main
    • 推送现有本地仓库
    git init --initial-branch=main
git remote add origin https://jihulab.com/1987/mydocs.git
git add .
git commit -m "Initial commit"
git push -u origin main
    • 推送现有远程仓库
    git remote rename origin old-origin
git remote add origin https://jihulab.com/1987/mydocs.git
git push -u origin --all
git push -u origin --tags
    • 获取add未commit修改
    git add
git diff --cached
    • 删除未忽略且已经push到远程的文件
    git rm -r --cached xxx
# 修改.gitignore
git add .
git commit -am "gitignore文件修改删除XXX文件"
