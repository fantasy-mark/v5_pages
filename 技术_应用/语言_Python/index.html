    动态声明

    from peewee import TextField, Model, SqliteDatabase

db = SqliteDatabase(DB_PATH)


def dynamic_declara(items):
    """
    用于动态声明创建peewee表类, 以下代码动态声明了一个FA_Report类, 使用如下:
    report_tab = dynamic_declara(titles)
    report_tab.select()     # FA_Report 继承了 peewee.Model 的select方法

    class FA_Report(Model):
        class Meta:
            database = db

        col1 = TextField()
        col2 = TextField()
        colN = TextField(unique=True) 

    :param items: 表项名称
    :return:
    """
    class_dict = {'Meta': type('Meta', (object,), {'database': db})}
    for column in items[:-1]:
        class_dict[column] = TextField(column_name=column)
    class_dict[items[-1]] = TextField(column_name=items[-1], unique=True)

    return type('FA_Report', (Model,), class_dict)

    运行时进入调试

    比如运行 mkdocs build 需要进入调试模式查看信息

    def a_func():
    //....此处停下
    import pdb
    pdb.set_trace()

    (venv) D:\pythonProject\flask\mydocs\my-project>mkdocs build
> d:\pythonproject\flask\mydocs\mkdocs\config\base.py(350)load_config() 
-> for key, value in options.copy().items():
(Pdb)

    当程序运行到 pdb.set_trace() 处时，它会暂停执行，并进入调试模式。你会看到一个命令提示符，可以在其中输入命令来控制调试器的行为。

    下面是一些常用的 pdb 命令：

    • n：执行下一行代码。
    • c：继续执行程序，直到遇到下一个断点或程序结束。
    • q：退出调试器并终止程序。
    • p <expression>：打印表达式的值。
    • pp <expression>：以更易读的方式打印表达式的值。
    • s：单步执行代码，如果遇到函数调用则进入函数内部。
    • h：显示帮助信息。

    是的，pdb 还有许多更高级的用法。下面是一些进阶技巧：

    • 条件断点：你可以使用 b 命令来设置条件断点。例如，b 10, x > 0 会在第 10 行设置一个断点，只有当 x > 0 时才会触发。
    • 查看堆栈：你可以使用 w 命令来查看当前的调用堆栈。
    • 上下文切换：你可以使用 u 和 d 命令来在调用堆栈中向上或向下移动。这样你就可以查看不同堆栈帧中的变量。
    • 跳过代码：你可以使用 j 命令来将程序计数器跳转到指定的行。这样你就可以跳过一些不感兴趣的代码。

    bt 命令是 pdb 中的一个命令，它用于打印当前线程的回溯信息（backtrace）。回溯信息显示了当前调用堆栈中的所有帧，以及每个帧中的源代码行号和函数名。

    要使用 bt 命令，只需在 pdb 的命令提示符下输入 bt 并按回车键即可。例如：

    import pdb

def foo():
    bar()

def bar():
    pdb.set_trace()

foo()

    当程序运行到 pdb.set_trace() 处时，它会暂停执行并进入调试模式。此时你可以输入 bt 命令来查看当前的回溯信息：

    > <ipython-input-3-6a3f8d2b8e5d>(8)bar()
-> pdb.set_trace()
(Pdb) bt
  <ipython-input-4-1f0c5a6b7a9c>(1)<module>()
-> foo()
  <ipython-input-3-6a3f8d2b8e5d>(5)foo()
-> bar()
> <ipython-input-3-6a3f8d2b8e5d>(8)bar()
-> pdb.set_trace()

    在这个例子中，回溯信息显示了当前调用堆栈中有三个帧：最底层的帧是 foo 函数，它调用了 bar 函数；bar 函数又调用了 pdb.set_trace() 函数。

    安装命令到系统

    # scirpt.py
import click


def common_options(f):
    f = click.option('--debug/--no-debug', default=False)(f)
    return f


@click.group()
def cli():
    pass


@cli.command()
@common_options
def command1(debug):
    if debug:
        click.echo('Debug mode is on')
    else:
        click.echo('Debug mode is off')


@cli.command()
@common_options
def command2(debug):
    if debug:
        click.echo('Debug mode is on')
    else:
        click.echo('Debug mode is off')


if __name__ == "__main__":
    cli()

    # setup.py 这个文件名不要改
from setuptools import setup, find_packages

setup(
    name='script',
    version='0.1',
    packages=find_packages(),
    include_package_data=True,
    install_requires=[
        'Click',
    ],
    entry_points='''
        [console_scripts]
        ccode=script:cli        # 定义命令 ccode -> script.cli
    ''',
)
