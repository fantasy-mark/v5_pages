    frida

    安装python & nodejs

    sudo apt install python
sudo update-alternatives --install /usr/bin/python python /usr/bin/python3.10 1
python -m pip install --user virtualenv virtualenvwrapper

    vi ~/.bashrc
export VIRTUALENVWRAPPER_PYTHON=/usr/bin/python3
export WORKON_HOME=~/.virtualenvs
source /usr/share/virtualenvwrapper/virtualenvwrapper.sh

    mkvirtualenv frida_env
# 下次使用
workon frida_env

    安装friad & frida_serve

    pip install frida -i https://pypi.mirrors.ustc.edu.cn/simple/
pip install frida-tools -i https://pypi.mirrors.ustc.edu.cn/simple/
frida --version
    下载对应版本 frida-serve 
    adb push xx-frida-server-xx /data/frida-serve
chmod 777 /data/frida-serve

    试用 frida

    npm i @types/frida-gum

    # android端
/data/frida-serve &
# or
/data/frida-serve -l 0.0.0.0:port

    # 列出目标设备的所有进程
frida-ps -U -a
frida-ps -H serve_ip:port -a

# -f 启用 spawn 模式 否则 启用 attach 模式
frida -U [-f] pkg_or_ps_name -l hello.js
# or
friad -H serve_ip:port [-f] ps_napkg_or_ps_name -l hello.js

        console.log('hello ' + msg)
    });
}

    强辅助 objection hook

    pip install objection

    # 注入进程
objection -g pkg_name explore

    # 枚举已有类
android hooking list classes
# 匹配搜索已有类
android hooking search classes @class_name_pattern
# 枚举指定类非构造方法签名
android hooking list class_methods @class_name
# hook所有非构造函数
android hooking watch class @class_name
# hook指定类指定方法
android hooking watch class_method @class_method @overload @option
# 如hook一个参数类型是int数组的函数
android hooking watch class_method xxx.xxxClass.xxxMethod "[I" --dump-args --dump-return --dump-backtrace
    Hook快速定位关键类、方法

    • objection 
      # 找出包内类
objection -g top.xjunz.tasker explore
 [usb] # android hooking search classes top.xjunz.tasker
Note that Java classes are only loaded when they are used, so if the expected class has not been found, it might not have been loaded yet.

# 复制所有类到 hook.txt 文件
# 在所有类前加上 android hooking watch class
objection -g top.xjunz.tasker explore -c hook.txt 

      # 列出加载的 libssl.so 函数名和映射地址, 用于 native hook
