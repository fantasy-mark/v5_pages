    • main.js导入Element plus
    import './assets/main.css'

import { createApp } from 'vue'
import App from './App.vue'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import zhCn from 'element-plus/es/locale/lang/zh-cn'

const app = createApp(App)
app.use(ElementPlus, {
    size: 'small',
    zIndex: 3000,
    locale: zhCn
})

app.mount('#app')

    使用axios

    • 前端代码
    // request.js
// npm install axios
import axios from 'axios'
const request = axios.create({
    baseURL: 'http://localhost:5432/api', // 基础路径,将统一的部分全部封装, 5432为flask端口
    withCredentials: true, // 表示请求可以携带cookie
})
//前端采用export.default，在写后端代码时用module.export
export default request

    //使用, flask需要配置跨域访问
request.get('/hello')
  .then(response => {
    console.log(response.data['message']);
  })
  .catch(error => {
  console.error(error);
});
    • 后端代码
    # pip install flask-cors
from flask_cors import CORS

bp = Blueprint('analyse', __name__, url_prefix='/api')

@bp.route('/hello')
def markdown():
    return jsonify({'message': 'hello'})

def create_app(test_config=None):
    app = Flask(__name__, instance_relative_config=True)
    # http://localhost:5173为vue服务
    CORS(app, origins='http://localhost:5173', methods=['GET', 'POST', 'PUT', 'DELETE'], allow_headers='Content-Type', supports_credentials=True)

    app.register_blueprint(bp)

    return app

flask = create_app()

if __name__ == '__main__':
    flask.run(host='0.0.0.0', port=5432)
