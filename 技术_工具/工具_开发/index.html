    neo4j

    1. 下载community版本

    2. 设置环境变量

    3. NEO4J_HOME : E:\neo4j-community-5.6.0

    4. PATH : %NEO4J_HOME%\bin

    使用

    1. 启动neo4j
    neo4j.bat console

    1. 浏览器登录

    2. 默认账号密码 neo4j，初次登录需要更改密码

    3. 测试是否生成节点

    create(e:Person{name:"李四"}) return e

    进阶-cypher语法

    CREATE
    -- 节点创建
CREATE (oliver:Person {name: 'Oliver Stone'}) 
CREATE (WallStreet:Movie {title: 'Wall Street'}) 

-- 创建关系 
CREATE (oliver)-[:DIRECTED]->(WallStreet)
    MATCH
    -- 带标签节点查询
MATCH (movie:Movie)
RETURN movie.title

-- 带标签节点查询, 与'Oliver Stone'相关的电影名
MATCH (:Person { name: 'Oliver Stone' })--(movie:Movie)
RETURN movie.title

-- 指定关系类型，ACTED_IN是创建节点时自定义的关系
MATCH (wallstreet:Movie { title: 'Wall Street' })<-[:ACTED_IN]-(actor)
RETURN actor.name
    • shortestPath最短路径,带约束的最短路径，使用WHERE
    -- Single shortest path with predicates
-- 不考虑father/son 的关系
MATCH (charlie:Person { name: 'Charlie Sheen' }),(martin:Person { name: 'Marlin Sheen' }), p = shortestPath((charlie)-[*]-(martin))
WHERE NONE (r IN relationships(p) WHERE type(r)= 'FATHER')
RETURN p
    MERGE
    -- MERGE确保模式在graph中存在，要么已经存在要么创建
-- MERGE带标签和属性的节点
MERGE (michael:Person { name: 'Michael Douglas' })
RETURN michael.name, michael.bornIn
    USE
    • 指定那个graph将被查询
    LOAD CSV
    • 用于导入csv文件
    • FROM后面接着csv的文件路径，用AS指定给变量
    • 可以从HTTPS、HTTP、FTP导入CSV
    -- 即从csv读取数据时每读取500行时存入数据一次
USING PERIODIC COMMIT 500
LOAD CSV FROM 'https://neo4j.com/docs/cypher-manual/4.1/csv/artists.csv' AS line
CREATE (:Artist { name: line[1], year: toInteger(line[2])})
